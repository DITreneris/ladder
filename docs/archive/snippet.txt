        // Initialize Audio Context Lazily
        function initAudio() {
            if (!audioCtx) {

        // Custom audio play helper (pure web oscillators)
        function playTone(freq, type, duration) {
            if (isMuted) return;
            try {
                initAudio();
                const osc = audioCtx.createOscillator();
                const gain = audioCtx.createGain();
                
                osc.type = type || 'sine';
                osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
                
                gain.gain.setValueAtTime(0.08, audioCtx.currentTime);
                gain.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + duration);
                
                osc.connect(gain);
                gain.connect(audioCtx.destination);
                
                osc.start();
                osc.stop(audioCtx.currentTime + duration);
            } catch (e) {
                console.warn("Audio Context blocked or failed:", e);
            }
        }

        // Random Rung Maker (Obstacle generator)
        function generateRung(forceEmpty = false) {
            let obstacle = null;
            let type = null;
            let coffee = null;

            if (!forceEmpty) {
                const rand = Math.random();
                
                // 30% Chance of an upcoming obstacle on higher levels
                if (rand < 0.35) {
                    obstacle = Math.random() < 0.5 ? 'left' : 'right';
                    
                    // Distribute dynamic obstacle types
                    const typeRand = Math.random();
                    if (typeRand < 0.5) {
                        type = 'meeting'; // Standard static blockage
                    } else if (typeRand < 0.8) {
                        type = 'reorg';    // Volatile swapping blockage
                    } else {
                        type = 'burnout';  // Standard grey cloud
                    }
                } else if (rand > 0.85) {
                    // Spawn delicious coffee replenishment cup on the opposite side of empty
                    coffee = Math.random() < 0.5 ? 'left' : 'right';
                }
            }

            const rung = {
                id: nextRungId++,
                obstacle: obstacle,
                type: type,
                coffee: coffee
            };
            return rung;
        }

        // Start Game State
        function startGame() {
            initAudio();
            score = 0;
            timeLeft = 100;
            isPlaying = true;
            isGameOverState = false;
            playerSide = 'left';
            nextRungId = 0;
            currentRank = 'Intern';
            
            // Generate clean base rungs
            rungs = [];
            // Bottom 3 are always empty for starting safety
            rungs.push(generateRung(true));
            rungs.push(generateRung(true));
            rungs.push(generateRung(true));
            
            for (let i = 3; i < MAX_VISIBLE_RUNGS; i++) {
                rungs.push(generateRung());
            }

            // Sync HTML displays
            document.getElementById('gameYearsLabel').innerText = "0.0";
            document.getElementById('burnoutMeter').style.width = "100%";
            document.getElementById('burnoutPercentLabel').innerText = "100%";
            document.getElementById('tapPrompt').classList.remove('hidden');
            
            updatePlayerVisualPosition();
            updateRankBadgeAndLabel();
            renderRungs();
            
            switchTab('game');
            stopGameLoops();

            // Run Burnout Meter Depletion Loop
            // Drain speed scales up gradually based on current years survived
            timerInterval = setInterval(() => {
                if (!isPlaying) return;
                
                // Math difficulty scaling:
                const years = Math.floor(score / 4);
                const drainRate = 1.0 + (years * 0.12); // Faster drain as career longevity increases
                
                timeLeft -= drainRate;
                if (timeLeft <= 0) {
                    timeLeft = 0;
                    triggerGameOver("Burnout", "Collant overload. Energy reserves fully exhausted before reaching promotion.");
                }

                // Warn user when burnout level gets critically high
                const meter = document.getElementById('burnoutMeter');
                meter.style.width = timeLeft + "%";
                document.getElementById('burnoutPercentLabel').innerText = Math.round(timeLeft) + "%";

                if (timeLeft < 25) {
                    meter.className = "h-full bg-red-600 rounded-full transition-all duration-75 animate-pulse";
                    if (Math.random() < 0.2) playTone(400, 'triangle', 0.05); // stress ticking audio
                } else {
                    meter.className = "h-full bg-gradient-to-r from-emerald-500 via-amber-500 to-red-500 rounded-full transition-all duration-75";
                }
            }, 100);

            // Run Reorganization dynamic obstacle swapper
            // Moves 'reorg' blocks dynamically every 600ms to confuse climbers
            reorgInterval = setInterval(() => {
                if (!isPlaying) return;
                let altered = false;
                rungs.forEach(r => {
                    if (r.type === 'reorg') {
                        r.obstacle = r.obstacle === 'left' ? 'right' : 'left';
                        altered = true;
                    }
                });
                if (altered) {
                    renderRungs();
                    // Play swift mechanical shuffle sound
                    if (Math.random() < 0.3) playTone(300, 'sine', 0.04);
                }
            }, 600);
        }

        // Clean up timing threads
        function stopGameLoops() {
            if (timerInterval) clearInterval(timerInterval);
            if (reorgInterval) clearInterval(reorgInterval);
        }

        // Handle Tap Action (Left or Right trigger)
        function handleTap(side) {
            if (!isPlaying || isGameOverState) return;

            // Remove visual guidelines prompt once they make their first climb move
            document.getElementById('tapPrompt').classList.add('hidden');

            playerSide = side;
            updatePlayerVisualPosition();

            // Determine if upcoming rung blocks current side
            // Note: Timberman logic checks collision against upcoming log falling onto current step
            const nextRung = rungs[1]; 
            
            if (nextRung && nextRung.obstacle === playerSide) {
                // Determine satirical game over message matching obstacle type
                let cause = "Reorganization";
                let detail = "A massive department restructuring shuffled you out of direct reports.";
                
                if (nextRung.type === 'meeting') {
                    cause = "Meeting Overload";
                    detail = "Fell into an all-hands call on slide layout. Reached maximum agenda tolerance.";
                } else if (nextRung.type === 'burnout') {
                    cause = "Burnout Flare";
                    detail = "Waded headfirst into a pile of corporate metrics with zero active coffee left.";
                }
                
                triggerGameOver(cause, detail);
                return;
            }

            // SUCCESSFUL STEP!
            score++;
            playTapSound();

            // Consume Coffee if it sits on our target side
            if (nextRung && nextRung.coffee === playerSide) {
                timeLeft = Math.min(100, timeLeft + 25); // Recover 25% burnout tolerance
                playCoffeeSound();
                showToast("+25% Energy Recovery! ☕");
            } else {
                // Standard climbing step recovers 1.5% energy passively
                timeLeft = Math.min(100, timeLeft + 1.5);
            }

            // Recycle Rung stack
            rungs.shift(); // Remove base rung
            rungs.push(generateRung()); // Feed new top rung
            
            // Refresh visuals
            const yearsSurvived = (score / 4).toFixed(1);
            document.getElementById('gameYearsLabel').innerText = yearsSurvived;
            
            checkPromotions();
            renderRungs();
        }

            climber.style.left = targetLeft;
        }

        // Evaluate promotions and scale visual rank attributes

            const years = Math.floor(score / 4);
            let prevRank = currentRank;

            if (years >= 35) {
                currentRank = 'CEO';
            } else if (years >= 10) {
                currentRank = 'Manager';
            } else {
                currentRank = 'Intern';
            }

            if (currentRank !== prevRank) {
                playPromoSound();
                displayPromotionBanner();
                updateRankBadgeAndLabel();
            }
        }

        function updateLeaderboardData(name, score, rank) {
            let savedBoard = localStorage.getItem('corp_ladder_boardroom');
            let board = [];
            
            if (savedBoard) {
                board = JSON.parse(savedBoard);
            } else {
                // Seed baseline office characters for healthy corporate competition
                board = [
                    { name: "Chad (Product VP)", score: 48.0, rank: "CEO", self: false },
                    { name: "Jeff (Founding Architect)", score: 38.5, rank: "CEO", self: false },
                    { name: "DevDave (Burnt Senior)", score: 26.0, rank: "Manager", self: false },
                    { name: "Silly Intern Sam", score: 4.5, rank: "Intern", self: false },
                    { name: "CorporateAI Bot", score: 18.0, rank: "Manager", self: false }
                ];
            }

            // Check if player already exists in mock list
            const existingIdx = board.findIndex(p => p.self === true || p.name === name);
            if (existingIdx !== -1) {
                if (score > board[existingIdx].score) {
                    board[existingIdx].score = score;
                    board[existingIdx].rank = rank;
                    board[existingIdx].name = name; // Update name in case they renamed
                }
            } else {
                board.push({ name: name, score: score, rank: rank, self: true });
            }

            // Re-order by years survived
            board.sort((a, b) => b.score - a.score);
            localStorage.setItem('corp_ladder_boardroom', JSON.stringify(board));
        }
